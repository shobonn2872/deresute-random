    function pickRandom(count) {

      if (count > songs.length) count = songs.length;

      const selected = [];

      const used = new Set();

      while (selected.length < count) {

        const idx = Math.floor(Math.random() * songs.length);

        if (!used.has(idx)) {

          used.add(idx);

          selected.push(`${selected.length + 1}. ${songs[idx]}`);

        }

      }

      document.getElementById("result").textContent = selected.join("\n");

    }
